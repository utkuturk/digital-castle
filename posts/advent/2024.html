Part 1

read_data <- function(path) {
  readLines(path) |>
    strsplit(" ") |>
    lapply(as.numeric) |>
    {
      \(rows) {
        max_length <- max(sapply(rows, length))
        lapply(rows, \(x) c(x, rep(NA, max_length - length(x))))
      }
    }() |>
    do.call(what = rbind) |>
    as.data.frame()
}


safe <- function(row) { # to be used with apply functions/iterate over rows
  row <- row[!is.na(row)]
  increasing <- all(diff(row) > 0)
  decreasing <- all(diff(row) < 0)
  difference <- all(abs(diff(row)) >= 1 & abs(diff(row)) <= 3)
  (increasing || decreasing) && difference
}

df <- read_data("2024/02/input")
df$safe <- apply(df, 1, safe)
sum(df$safe)
[1] 624

safe <- function(row) { # to be used with apply functions/iterate over rows
  row <- row[!is.na(row)]
  increasing <- all(diff(row) > 0)
  decreasing <- all(diff(row) < 0)
  difference <- all(abs(diff(row)) >= 1 & abs(diff(row)) <= 3)
  (increasing || decreasing) && difference
}


process_row <- function(row) {
  original_row <- row
  indices_to_check <- seq_along(row)
  tried_indices <- integer(0)

  if (safe(row)) {return(TRUE)}

  while (length(indices_to_check) > 0) {
    remove_index <- sample(indices_to_check, 1)

    row_with_na <- row
    row_with_na[remove_index] <- NA

    if (safe(row_with_na)) { return(TRUE) }

    tried_indices <- c(tried_indices, remove_index)
    indices_to_check <- setdiff(seq_along(row), tried_indices)
  }
  return(FALSE)
}

check_safety <- function(df) {
  df$safe <- apply(df, 1, process_row)
  return(df)
}

df <- read_data("2024/02/input")
# df$safe <- apply(df, 1, safe)
df <- check_safety(df)
